## features

* one step look ahead to play a turn 
```
 def play(self):
        visited = []
        self.update_canvas()
        valid_moves = self.valid_moves_f()
        for move in valid_moves:
            if move not in visited:
                board = self.board.copy()
                board[move[0]][move[1]] = self.agent_mark
                visited.append(move)
                out = self.game.game_conditions(board)
                if out == 1:
                    return move
        if not valid_moves:
            return -1
        elif len(valid_moves) == 1:
            return valid_moves[0]
        else:
            return valid_moves[np.random.choice([*range(len(valid_moves)-1)])]
```

```
    def op_play(self):
         visited = []
         self.update_canvas()
         valid_moves = self.valid_moves_f()
         for move in valid_moves:
             if move not in visited:
                 board = self.board.copy()
                 board[move[0]][move[1]] = self.op_mark
                 visited.append(move)
                 out = self.game.game_conditions(board)
                 if out == 1:
                     return move
         if not valid_moves:
             return -1
         elif len(valid_moves) == 1:
             return -1
         else:
             return -1
```
